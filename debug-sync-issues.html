🔍 Sync Debug Tool

This tool helps debug unsynced objects in your POS system.

Instructions

  1. Open your POS application in another tab
  2. Open browser developer tools (F12)
  3. Go to the Console tab
  4. Copy and paste the script below into the console
  5. Press Enter to run the debug analysis

Debug Script

Copy this script and run it in your browser console:

// Enhanced sync debug script
async function debugSyncIssues() {
  console.log('🔍 Starting comprehensive sync debug analysis...');
  
  // Check if we can access the database
  const db = window.db || (window.OfflineDataContext && window.OfflineDataContext.db);
  if (!db) {
    console.error('❌ Cannot access database. Make sure the app is loaded and try again.');
    return;
  }
  
  const tableNames = [
    'stores', 'products', 'suppliers', 'customers', 'cash_drawer_accounts',
    'inventory_bills', 'inventory_items', 'transactions', 'bills',
    'bill_line_items', 'bill_audit_logs', 'cash_drawer_sessions'
  ];
  
  const results = {};
  let totalUnsynced = 0;
  const validationErrors = [];
  
  console.log('📊 Checking each table for unsynced records...');
  
  for (const tableName of tableNames) {
    try {
      const table = db[tableName];
      if (!table) {
        console.log(`⚠️ Table ${tableName} not found`);
        continue;
      }
      
      const unsyncedRecords = await table.filter(item => !item._synced).toArray();
      const count = unsyncedRecords.length;
      
      if (count > 0) {
        results[tableName] = {
          count,
          records: unsyncedRecords
        };
        totalUnsynced += count;
        
        console.log(`📋 ${tableName}: ${count} unsynced records`);
        
        // Validate each record
        for (const record of unsyncedRecords) {
          const validation = await validateRecord(tableName, record, db);
          if (!validation.valid) {
            validationErrors.push({
              table: tableName,
              recordId: record.id,
              errors: validation.errors
            });
          }
        }
      }
    } catch (error) {
      console.error(`❌ Error checking ${tableName}:`, error);
    }
  }
  
  console.log(`\n📈 Summary:`);
  console.log(`Total unsynced records: ${totalUnsynced}`);
  console.log(`Validation errors: ${validationErrors.length}`);
  
  if (validationErrors.length > 0) {
    console.log('\n❌ Validation Errors Found:');
    validationErrors.forEach(error => {
      console.log(`  ${error.table}.${error.recordId}: ${error.errors.join(', ')}`);
    });
  }
  
  // Check sync service status
  console.log('\n🔄 Checking sync service status...');
  try {
    const syncService = window.syncService || (window.OfflineDataContext && window.OfflineDataContext.syncService);
    if (syncService) {
      console.log('Sync service is available');
      if (syncService.isCurrentlyRunning) {
        console.log('Sync is currently running');
      } else {
        console.log('Sync is not running');
      }
    } else {
      console.log('Sync service not accessible');
    }
  } catch (error) {
    console.error('Error checking sync service:', error);
  }
  
  return { results, validationErrors, totalUnsynced };
}

async function validateRecord(tableName, record, db) {
  const errors = [];
  
  try {
    switch (tableName) {
      case 'inventory_items':
        if (!record.product_id) errors.push('Missing product_id');
        if (!record.supplier_id) errors.push('Missing supplier_id');
        if (record.quantity < 0) errors.push('Negative quantity');
        if (record.quantity === undefined || record.quantity === null) errors.push('Missing quantity');
        
        // Check if referenced product exists
        if (record.product_id) {
          const product = await db.products.get(record.product_id);
          if (!product) errors.push(`Referenced product ${record.product_id} not found`);
        }
        
        // Check if referenced supplier exists
        if (record.supplier_id) {
          const supplier = await db.suppliers.get(record.supplier_id);
          if (!supplier) errors.push(`Referenced supplier ${record.supplier_id} not found`);
        }
        
        // Check if referenced batch exists (if provided)
        if (record.batch_id) {
          const batch = await db.inventory_bills.get(record.batch_id);
          if (!batch) errors.push(`Referenced batch ${record.batch_id} not found`);
        }
        break;
        
      case 'bill_line_items':
        if (!record.bill_id) errors.push('Missing bill_id');
        if (!record.product_id) errors.push('Missing product_id');
        if (!record.supplier_id) errors.push('Missing supplier_id');
        if (!record.quantity || record.quantity <= 0) errors.push('Invalid quantity');
        
        // Check if referenced bill exists
        if (record.bill_id) {
          const bill = await db.bills.get(record.bill_id);
          if (!bill) errors.push(`Referenced bill ${record.bill_id} not found`);
        }
        
        // Check if referenced product exists
        if (record.product_id) {
          const product = await db.products.get(record.product_id);
          if (!product) errors.push(`Referenced product ${record.product_id} not found`);
        }
        
        // Check if referenced supplier exists
        if (record.supplier_id) {
          const supplier = await db.suppliers.get(record.supplier_id);
          if (!supplier) errors.push(`Referenced supplier ${record.supplier_id} not found`);
        }
        break;
        
      case 'bills':
        if (!record.bill_number) errors.push('Missing bill_number');
        if (!record.status) errors.push('Missing status');
        if (record.total_amount === undefined || record.total_amount === null) errors.push('Missing total_amount');
        break;
        
      case 'transactions':
        if (!record.amount) errors.push('Missing amount');
        if (!record.type) errors.push('Missing type');
        break;
    }
  } catch (error) {
    errors.push(`Validation error: ${error.message}`);
  }
  
  return {
    valid: errors.length === 0,
    errors
  };
}

// Make functions available globally
window.debugSyncIssues = debugSyncIssues;
window.validateRecord = validateRecord;

console.log('🔧 Enhanced debug script loaded. Run debugSyncIssues() to analyze sync issues.');

Quick Fixes

If you find validation errors, here are some quick fixes:

  • Missing foreign keys: The referenced record might have been deleted. Check if the parent record exists.
  • Negative quantities: This might be from a calculation error. Check the inventory deduction logic.
  • Missing required fields: The record might be incomplete. Check the creation logic.

Manual Sync Trigger

If you want to manually trigger a sync after fixing issues:

// In browser console:
// Get the offline data context
const offlineData = window.OfflineDataContext || document.querySelector('[data-offline-context]');
if (offlineData && offlineData.sync) {
  offlineData.sync().then(result => {
    console.log('Manual sync result:', result);
  });
} else {
  console.log('Cannot access sync function. Make sure the app is fully loaded.');
}
